#include <iostream>
#include <iomanip> // Untuk format output
using namespace std;

int main() {
    string namaBarang;
    int jumlah;
    double hargaSatuan, total;

    cout << "Masukkan nama barang: ";
    cin >> namaBarang;
    cout << "Masukkan harga satuan: Rp";
    cin >> hargaSatuan;
    cout << "Masukkan jumlah barang: ";
    cin >> jumlah;

    total = hargaSatuan * jumlah;

    // Berikan diskon 10% jika total lebih dari Rp1.000.000
    if (total > 1000000) {
        total *= 0.9;
        cout << "Anda mendapatkan diskon 10%.\n";
    }

    cout << fixed << setprecision(2);
    cout << "Total harga untuk " << namaBarang << ": Rp" << total << endl;

    return 0;
}
